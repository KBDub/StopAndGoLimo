Blade Usage

{{-- Trigger anywhere on the page --}}
<x-ui.modal-trigger modal="confirm-delete">Delete Item</x-ui.modal-trigger>

{{-- Raw Alpine dispatch (works from any element) --}}
<button @click="$dispatch('open-modal',{name:'confirm-delete'})">Open</button>

{{-- Livewire: $this->dispatch('open-modal', name: 'confirm-delete'); --}}

{{-- Modal definition --}}
<x-ui.modal
    name="confirm-delete"
    title="Confirm Delete"
    variant="danger"
    size="sm"
>
    <p>Are you sure? This cannot be undone.</p>

    <x-slot:footer>
        <x-ui.modal-trigger modal="confirm-delete" as="close"
            class="t5-modal-btn t5-modal-btn-ghost">Cancel</x-ui.modal-trigger>
        <button wire:click="delete" class="t5-modal-btn t5-modal-btn-danger">Delete</button>
    </x-slot:footer>
</x-ui.modal>
